const ResetPasswordEmailScreen = () => {

        <h2 className="text-3xl font-bold text-center mb-8">

              >
                Resend Email
              </button>
            </div>
          </form>
        </div>
      </div>
    </div>
  );
};

export default ResetPasswordEmailScreen;
